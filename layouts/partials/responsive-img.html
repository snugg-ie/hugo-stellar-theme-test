{{- /* Rewrites micro.blog-hosted JPG in post content to a small/medium srcset. micro.blog can resize .jpg via -s (300px) and -m (~1000px) variants but NOT .webp, so .webp and external images are left untouched (graceful during webp->jpg migration). */ -}} {{- /* Strip any existing micro.blog size suffix (-s/-m/-xl) so we always rebuild from the base name; otherwise abc-xl.jpg would become abc-xl-s.jpg, which doesn't exist. */ -}} {{- return (. | replaceRE `src="(https://(?:snugg\.ie/uploads|cdn\.uploads\.micro\.blog)/[^"]+?)(?:-xl|-m|-s)?\.(jpe?g)"` `src="${1}-s.${2}" srcset="${1}-s.${2} 300w, ${1}-m.${2} 1000w" sizes="(min-width: 571px) 32vw, 100vw"`) -}}
