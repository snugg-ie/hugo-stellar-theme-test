{{- /* Generates a 1200x630 OpenGraph card for a post and returns the image resource. Precedence: title > summary > image. - title: big 88px headline + byline - summary: smaller 48px text + byline (titleless/status posts) - short summary + image(s): caption on top, photo(s) shrink below - image(s): rounded photo / collage, no byline Non-image media (e.g. video) in .Params.images is skipped (no .Fill). */ -}} {{- $base := resources.Get "og/base.png" -}} {{- $fontLight := resources.Get "fonts/SourceSans3-Light.ttf" -}} {{- $fontSemi := resources.Get "fonts/SourceSans3-Semibold.ttf" -}} {{- $avatar := resources.Get "og/avatar.png" -}} {{- $logoShadow := resources.Get "og/logo-shadow.png" -}} {{- $logoPlain := resources.Get "og/logo-plain.png" -}} {{- /* swap toggle: site param, overridable per-post */ -}} {{- $swap := false -}} {{- with .Site.Params.og }}{{ $swap = (.swapLogoAvatar | default false) }}{{ end -}} {{- if isset .Params "swap_logo_avatar" }}{{ $swap = .Params.swap_logo_avatar }}{{ end -}} {{- $date := .Date.Format "Mon 2 Jan, 2006" -}} {{- $isTitle := ne .Title "" -}} {{- $sum := trim (replaceRE "\\s+" " " (.Summary | plainify | htmlUnescape)) " " -}} {{- $hasSum := ne $sum "" -}} {{- /* keep only real images — a video (etc.) in .Params.images has no .Fill and breaks the build */ -}} {{- $imgs := slice -}} {{- range (.Params.images | default (slice)) -}} {{- if findRE `(?i)\.(jpe?g|png|gif|webp)(\?|$)` . -}}{{- $imgs = $imgs | append . -}}{{- end -}} {{- end -}} {{- $n := len $imgs -}}{{- if gt $n 4 }}{{ $n = 4 }}{{ end -}} {{- $captionMode := and (not $isTitle) (gt $n 0) $hasSum (le (len $sum) 38) -}} {{- $imageMode := and (not $isTitle) (gt $n 0) (not $hasSum) -}} {{- $card := $base -}} {{- if or $captionMode $imageMode -}} {{- /* ===== IMAGE CARD (optionally captioned), no byline ===== */ -}} {{- $mBig := resources.Get "og/mask-big.png" -}} {{- $mMed := resources.Get "og/mask-med.png" -}} {{- $mSm := resources.Get "og/mask-small.png" -}} {{- $card = resources.Get "og/base-plain.png" -}} {{- /* crop focus: "smart" (default) or a named anchor via front matter og_anchor */ -}} {{- $anchor := "smart" -}}{{- if isset .Params "og_anchor" }}{{ $anchor = .Params.og_anchor }}{{ end -}} {{- /* photo area shrinks (pushed down) when there's a caption on top */ -}} {{- $top := 50 -}}{{- $ih := 530 -}}{{- $rh := 257 -}}{{- $row2 := 323 -}} {{- if $captionMode -}}{{- $top = 142 -}}{{- $ih = 433 -}}{{- $rh = 207 -}}{{- $row2 = 368 -}}{{- end -}} {{- $slots := slice -}} {{- if eq $n 1 -}} {{- $slots = slice (dict "x" 50 "y" $top "w" 1100 "h" $ih "m" $mBig) -}} {{- else if eq $n 2 -}} {{- $slots = slice (dict "x" 50 "y" $top "w" 542 "h" $ih "m" $mMed) (dict "x" 608 "y" $top "w" 542 "h" $ih "m" $mMed) -}} {{- else if eq $n 3 -}} {{- $slots = slice (dict "x" 50 "y" $top "w" 542 "h" $rh "m" $mSm) (dict "x" 608 "y" $top "w" 542 "h" $rh "m" $mSm) (dict "x" 50 "y" $row2 "w" 542 "h" $rh "m" $mSm) -}} {{- else -}} {{- $slots = slice (dict "x" 50 "y" $top "w" 542 "h" $rh "m" $mSm) (dict "x" 608 "y" $top "w" 542 "h" $rh "m" $mSm) (dict "x" 50 "y" $row2 "w" 542 "h" $rh "m" $mSm) (dict "x" 608 "y" $row2 "w" 542 "h" $rh "m" $mSm) -}} {{- end -}} {{- range $i, $s := $slots -}} {{- $t := try (resources.GetRemote (absURL (index $imgs $i))) -}} {{- with $t.Value -}} {{- if eq .ResourceType "image" -}} {{- /* crop keeps the subject ($anchor); PNG so the rounded (masked) corners stay transparent */ -}} {{- $ph := (.Fill (printf "%dx%d %s png" $s.w $s.h $anchor)) | images.Filter (images.Mask $s.m) -}} {{- $card = $card | images.Filter (images.Overlay $ph $s.x $s.y) -}} {{- end -}} {{- end -}} {{- end -}} {{- if $captionMode -}} {{- $card = $card | images.Filter (images.Text $sum (dict "color" "#ffffff" "size" 40 "x" 56 "y" 54 "font" $fontLight)) -}} {{- end -}} {{- /* favicon-style sprinkles tucked into the top-right gutter */ -}} {{- $card = $card | images.Filter (images.Overlay ($logoPlain.Resize "52x") 1140 6) -}} {{- else -}} {{- /* ===== TEXT CARD: title (big) or summary (smaller) ===== */ -}} {{- $headline := cond $isTitle .Title $sum -}} {{- $size := 88 -}}{{- $max := 23 -}}{{- $maxLines := 2 -}}{{- $ty := 150 -}}{{- $tls := 7 -}} {{- if not $isTitle -}}{{- $size = 48 -}}{{- $max = 36 -}}{{- $maxLines = 4 -}}{{- $ty = 162 -}}{{- $tls = 12 -}}{{- end -}} {{- $lines := slice -}} {{- $cur := "" -}} {{- range $w := split $headline " " -}} {{- $tryline := cond (eq $cur "") $w (printf "%s %s" $cur $w) -}} {{- if le (len $tryline) $max -}} {{- $cur = $tryline -}} {{- else -}} {{- $lines = $lines | append $cur -}} {{- $cur = $w -}} {{- end -}} {{- end -}} {{- if ne $cur "" }}{{ $lines = $lines | append $cur }}{{ end -}} {{- $shown := $lines -}} {{- if gt (len $lines) $maxLines -}} {{- $keep := first $maxLines $lines -}} {{- $shown = (first (sub $maxLines 1) $keep) | append (printf "%s…" (index $keep (sub $maxLines 1))) -}} {{- end -}} {{- $title := delimit $shown "\n" | string -}} {{- if eq $title "" }}{{ $title = " " }}{{ end -}} {{- $trIcon := $logoShadow.Resize "130x" -}}{{- $trX := 998 -}}{{- $trY := 76 -}} {{- $blIcon := $avatar.Resize "76x" -}}{{- $blX := 90 -}}{{- $blY := 464 -}} {{- $nameX := 186 -}}{{- $nmY := 470 -}}{{- $stY := 506 -}} {{- if $swap -}} {{- $trIcon = $avatar.Resize "120x" -}}{{- $trX = 990 -}}{{- $trY = 84 -}} {{- $blIcon = $logoPlain.Resize "100x" -}}{{- $blX = 71 -}}{{- $blY = 458 -}} {{- $nameX = 172 -}}{{- $nmY = 475 -}}{{- $stY = 512 -}} {{- end -}} {{- $card = $base | images.Filter (images.Text $date (dict "color" "#cf8298" "size" 27 "x" 90 "y" 88 "font" $fontSemi)) (images.Text $title (dict "color" "#ffffff" "size" $size "x" 90 "y" $ty "linespacing" $tls "font" $fontLight)) (images.Text "Evie Snuggle" (dict "color" "#ffffff" "size" 30 "x" $nameX "y" $nmY "font" $fontSemi)) (images.Text "snugg.ie" (dict "color" "#c9bdc3" "size" 23 "x" $nameX "y" $stY "font" $fontLight)) (images.Overlay $trIcon $trX $trY) (images.Overlay $blIcon $blX $blY) -}} {{- end -}} {{- return $card -}}
